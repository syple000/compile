public class Test {
    static final public int CONST = 12;

    public void main() {
        TestClass testClass(22);
        int a = 1;
        int b = 2;
        a = a++ + b--;
        return testFunc(a, 12);
    }
    /* test func */
    public int testFunc(int a, float b) {
        if (a < b) {
            return CONST + 1;
        } else {
            if (a == b) {
                return CONST;
            }
            return CONST - 1;
        }
    } 
};

    TestClass(int a) {
        num = a;
    }
};
